export const PepList = ({ header = "Watchlist Sources", dataId = "pep-data", rows }) => {
  const { useState, useMemo, useEffect } = React; // React is in MDX global scope

  // Demo rows so preview always renders
  const DEMO_ROWS = [
    { "List Name": "OFAC SDN List", Country: "United States", Region: "North America", Tier: "Standard" },
    { "List Name": "Canada OSFI Entity List", Country: "Canada", Region: "North America", Tier: "Plus/Premium" },
    { "List Name": "EU Consolidated Sanctions", Country: "EU", Region: "Europe", Tier: "Plus/Premium" },
  ];

  function normalize(parsed) {
    if (!parsed) return [];
    if (Array.isArray(parsed)) {
      return parsed.map(r =>
        Array.isArray(r)
          ? { "List Name": r[0], Country: r[1], Region: r[2], Tier: r[3] }
          : {
              "List Name": r.l || r["List Name"] || "",
              Country: r.c || r.Country || "",
              Region: r.r || r.Region || "",
              Tier: r.t || r.Tier || "",
            }
      );
    }
    return [];
  }

  function readHidden(id) {
    if (typeof document === "undefined") return null;
    const el = document.getElementById(id);
    if (!el) return null;
    try {
      const raw = el.textContent || "";
      return JSON.parse(raw);
    } catch (e) {
      console.warn("PepList: parse failed", e);
      return null;
    }
  }

  const [rowData, setRowData] = useState(() =>
    rows ? normalize(rows) : DEMO_ROWS
  );

  useEffect(() => {
    if (rows) return; // prop wins
    const parsed = readHidden(dataId);
    if (parsed) setRowData(normalize(parsed));
  }, [rows, dataId]);

  const [activeRegion, setActiveRegion] = useState("All Regions");
  const [searchTerm, setSearchTerm] = useState("");

  const regionOptions = useMemo(() => {
    const seen = {};
    rowData.forEach(r => { if (r.Region) seen[r.Region] = true; });
    return ["All Regions"].concat(Object.keys(seen).sort());
  }, [rowData]);

  const filtered = useMemo(() => {
    const q = searchTerm.trim().toLowerCase();
    return rowData.filter(r => {
      if (activeRegion !== "All Regions" && r.Region !== activeRegion) return false;
      if (!q) return true;
      return (
        (r["List Name"] || "").toLowerCase().includes(q) ||
        (r.Country || "").toLowerCase().includes(q) ||
        (r.Region || "").toLowerCase().includes(q) ||
        (r.Tier || "").toLowerCase().includes(q)
      );
    });
  }, [rowData, activeRegion, searchTerm]);

  return (
    <div className="wl-root">
      <div className="wl-card">
        <div className="flex items-center mb-2">
          <i className="fa fa-solid fa-database mr-2 text-gray-500 dark:text-gray-300" />
          <span className="font-medium">{header}</span>
          <span className="ml-2 text-xs text-gray-500 dark:text-gray-300">(rows: {rowData.length})</span>
        </div>

        {/* Search */}
        <div className="wl-searchRow">
          <input
            className="wl-input"
            placeholder="Search…"
            value={searchTerm}
            onChange={e => setSearchTerm(e.target.value)}
          />
          {searchTerm ? (
            <button className="wl-clear" onClick={() => setSearchTerm("")}>Clear</button>
          ) : null}
        </div>

        {/* Regions */}
        <div className="wl-sectionTitle">Regions</div>
        <div className="wl-pills">
          {regionOptions.map(opt => (
            <button
              key={opt}
              className={"wl-pill" + (opt === activeRegion ? " is-active" : "")}
              onClick={() => setActiveRegion(opt)}
            >
              {opt}
            </button>
          ))}
        </div>

        {/* Table */}
        <div className="wl-meta">Showing {filtered.length} of {rowData.length}</div>
        <div className="rdmd-table">
          <table>
            <thead>
              <tr><th>List Name</th><th>Country</th><th>Region</th><th>Tier</th></tr>
            </thead>
            <tbody>
              {filtered.length ? filtered.map((r, i) => (
                <tr key={i}>
                  <td>{r["List Name"]}</td>
                  <td>{r.Country}</td>
                  <td>{r.Region}</td>
                  <td>{r.Tier}</td>
                </tr>
              )) : (
                <tr><td colSpan="4" className="wl-empty">No results</td></tr>
              )}
            </tbody>
          </table>
        </div>
      </div>
    </div>
  );
};
